1   package de.turnertech.measures;
2   
3   import java.util.Objects;
4   
5   /**
6    * <p>A Measure is a description of some quantity of a {@link Unit}, where a {@link Unit} is a standardised and
7    * accepted quantity, size, ammount, period or other kind of measurement. For example, a {@link Unit#METRE} 
8    * represents a standardised length.</p>
9    * 
10   * <p>This implementation stores an immutable {@link Unit} instance. To obtain a {@link Measure} with another
11   * {@link Unit} you must either explicitely instantiate a new instance, or convert the Measure using one of the
12   * either {@link #convertTo(Unit)} or the class {@link UnitConverter}</p>
13   */
14  public class Measure {
15      
16      private double quantity;
17  
18      private final Unit unit;
19      
20      /**
21       * Constructs an instance with the provided parameters.
22       * @param quantity The quantity of Units in this Measure. Mutable.
23       * @param unit The Unit of this Measure. This is immutable.
24       */
25      public Measure(final double quantity, final Unit unit) {
26          this.quantity = quantity;
27          this.unit = Objects.requireNonNull(unit, "Unit instance provided to Measure constructor may not be null");
28      }
29  
30      /**
31       * <p>Helper function for checking if the Measure is equal to within a supplied
32       * tolerance. This is usefull in several situations as the Double Precision
33       * mechanism can very quickly introduce innacuracies which prevent .equals() 
34       * from providing a useful result.</p>
35       * 
36       * <p>If the Measures do not suare the same base unit, a conversion will 
37       * first be attempted.</p>
38       * 
39       * @param other Measure to check against.
40       * @param tolerance allowed difference between the numbers.
41       * @return true or false.
42       */
43      public boolean equalsWithTolerance(final Measure other, final double tolerance) {
44          final Measure otherInThisUnit = UnitConverter.convert(other.quantity, other.unit, unit);
45          final double difference = Math.abs(this.quantity - otherInThisUnit.quantity);
46          return difference <= tolerance;
47      }
48  
49      /**
50       * Gets the quantity of Units stored in this Measure.
51       * @return the quantity of Units stored in this Measure.
52       */
53      public double getQuantity() {
54          return quantity;
55      }
56  
57      /**
58       * Sets the quantity of Units stored in this Measure.
59       * @param quantity the quantity of Units stored in this Measure.
60       */
61      public void setQuantity(final double quantity) {
62          this.quantity = quantity;
63      }
64  
65      /**
66       * Sets the Unit of this Measure.
67       * @return the Unit of this Measure.
68       */
69      public Unit getUnit() {
70          return unit;
71      }
72  
73      /**
74       * <p>Helper function to convert this Measure to a new Measure with the 
75       * supplied newUnit parameter. This function exists in place of a setter as
76       * the unit member is immutable.</p>
77       * 
78       * <p>Internally this function simply delegates to 
79       * {@link UnitConverter#convert(Measure, Unit)}</p>
80       * 
81       * @param newUnit to be used in the creation of the new Measure instance.
82       * @return a new Measure instance in the supplied Unit.
83       */
84      public Measure convertTo(final Unit newUnit) {
85          return UnitConverter.convert(this, newUnit);
86      }
87  
88      @Override
89      public int hashCode() {
90          int hash = 5;
91          hash = 61 * hash + (int) (Double.doubleToLongBits(this.quantity) ^ (Double.doubleToLongBits(this.quantity) >>> 32));
92          hash = 61 * hash + Objects.hashCode(this.unit);
93          return hash;
94      }
95  
96      @Override
97      public boolean equals(Object obj) {
98          if (this == obj) {
99              return true;
100         }
101         if (obj == null) {
102             return false;
103         }
104         if (getClass() != obj.getClass()) {
105             return false;
106         }
107         final Measure other = (Measure) obj;
108         if (Double.doubleToLongBits(this.quantity) != Double.doubleToLongBits(other.quantity)) {
109             return false;
110         }
111         return Objects.equals(this.unit, other.unit);
112     }
113     
114     /**
115      * We use the SI standard for the String representation. There is always a
116      * space between the quantity and unit, unless it is a planar angle.
117      * @return the Measure as a string formatted as per the SI Brochure section 
118      * 5.3.3
119      */
120     @Override
121     public String toString() {
122         if(unit == Unit.DEGREE || unit == Unit.MINUTE_ANGLE || unit == Unit.SECOND_ANGLE) {
123             return String.format("%f%s", quantity, unit.toString());
124         }
125         return String.format("%f %s", quantity, unit.toString());
126     }
127 
128 }
